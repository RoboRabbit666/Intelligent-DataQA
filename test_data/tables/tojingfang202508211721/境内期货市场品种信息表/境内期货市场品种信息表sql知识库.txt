问题：当日成交量前五的品种是哪些？ 
SELECT CLAS_NAME，trd_qty,fut_opt_flag
FROM 
    dataqa.fut_mrkt_clas_info_tab

    BIZ_DT = '2025-07-02'

    trd_qty DESC
LIMIT 5;

问题：2025年7月主力合约成交额占全市场比例是多少
SELECT 
    (SUM(main_comd_trd_amt) / SUM(trd_amt)) * 100 AS 主力合约成交额占比 
FROM dataqa.fut_mrkt_clas_info_tab

问题：找出2025-07月近月合约持仓量大于1000手的品种
SELECT 
    CLAS_NAME, 
    near_mon_comd, 
    sum(near_mon_comd_hld_qty) as 近月合约持仓量
FROM 
    dataqa.fut_mrkt_clas_info_tab 
WHERE 
    near_mon_comd_hld_qty > 1000
	AND BIZ_DT BETWEEN '2025-07-01' AND '2025-07-31'
group by CLAS_NAME, near_mon_comd 

问题：黄金期货近7天的成交额
SELECT BIZ_DT , trd_amt
FROM 
    dataqa.fut_mrkt_clas_info_tab
WHERE 
    CLAS_CODE = 'AU'  -- 黄金期货品种代码为'AU'
    AND fut_opt_flag = 'F'
    AND BIZ_DT > ('2025-07-02'::date - 7)  -- 近7天（含今日）
ORDER BY 
    BIZ_DT;

问题：2025年第二季度各交易所的总成交额是多少？

FROM dataqa.fut_mrkt_clas_info_tab

问题：2025年第三季度哪些品种发生过主力合约切换
SELECT DISTINCT CLAS_NAME 
FROM dataqa.fut_mrkt_clas_info_tab 
WHERE BIZ_DT BETWEEN '2025-07-01' AND '2025-09-30' 
GROUP BY CLAS_NAME, main_comd 
HAVING COUNT(DISTINCT main_comd) > 1;
