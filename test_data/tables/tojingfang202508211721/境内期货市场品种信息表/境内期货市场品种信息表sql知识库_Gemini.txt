问题：2025年7月，中国金融期货交易所的股指期货（分类代码以'I'开头）和国债期货（分类代码以'T'开头）的总成交额分别是多少？
SELECT
CASE
WHEN CLAS_CODE LIKE 'I%' THEN '股指期货'
WHEN CLAS_CODE LIKE 'T%' THEN '国债期货'
END AS category,
SUM(trd_amt) as total_trd_amt
FROM dataqa.fut_mrkt_clas_info_tab
WHERE BIZ_DT BETWEEN '2025-07-01' AND '2025-07-31' AND EXCH_CODE = 'CFFEX' AND (CLAS_CODE LIKE 'I%' OR CLAS_CODE LIKE 'T%')
GROUP BY category;
