问题：罗列不同会员有相同的联系人的会员信息？
SELECT a.*
FROM dataqa.mem_base_info_tab a
JOIN (
  SELECT ctc_name, COUNT(*) AS cnt
  FROM dataqa.mem_base_info_tab
    and biz_dt='2025-07-02'
	and exch_code='CZCE'
  GROUP BY ctc_name
  HAVING COUNT(*) > 1
) b ON a.ctc_name = b.ctc_name;
